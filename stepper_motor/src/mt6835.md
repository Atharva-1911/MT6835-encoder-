#include <Arduino.h>

#define MISO_PIN 25

void setup() {
    Serial.begin(115200);
    pinMode(MISO_PIN, INPUT);  // pure input, no pullup yet
}

void loop() {
    int val = digitalRead(MISO_PIN);
    Serial.print("MISO Pin 25 reads: ");
    Serial.println(val);
    delay(500);
}
STEP 1: Read the Miso pin at 25 pin number of esp32. The output reads 1 if the miso pin is connected properly.
